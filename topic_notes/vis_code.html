import matplotlib.lines as mlines

# Function to draw line segments
def newline(p1, p2):
ax = plt.gca()
l = mlines.Line2D([p1[0],p2[0]], [p1[1],p2[1]], color='#c2c0c0', alpha=0.5, linestyle='--', linewidth=3, zorder=1)
ax.add_line(l)
return l
